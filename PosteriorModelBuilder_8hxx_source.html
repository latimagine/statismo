117 template <typename T>
118 UniquePtrType<typename PosteriorModelBuilder<T>::StatisticalModelType>
119 PosteriorModelBuilder<T>::BuildNewModelFromModel(const StatisticalModelType * inputModel,
120  const PointValueWithCovarianceListType & pointValuesWithCovariance,
121  bool computeScores) const
122 {
123  STATISMO_LOG_INFO("Building new model from model");
124  const RepresenterType * representer = inputModel->GetRepresenter();
125 
126  const MatrixType & Q = inputModel->GetPCABasisMatrix();
127  const VectorType & mu = inputModel->GetMeanVector();
128 
129  // this method only makes sense for a proper PPCA model (e.g. the noise term is properly defined)
130  // if the model has zero noise, we assume a small amount of noise
131  double rho2 = std::max((double)inputModel->GetNoiseVariance(), (double)Superclass::sk_tolerance);
132  auto dim = representer->GetDimensions();
133 
134  // build the part matrices considering only the points that are fixed
135  auto numPrincipalComponents = inputModel->GetNumberOfPrincipalComponents();
136  MatrixType Q_g(pointValuesWithCovariance.size() * dim, numPrincipalComponents);
137  VectorType mu_g(pointValuesWithCovariance.size() * dim);
138  VectorType s_g(pointValuesWithCovariance.size() * dim);
139 
140  MatrixType LQ_g(pointValuesWithCovariance.size() * dim, numPrincipalComponents);
141 
142  unsigned i{ 0 };
143  for (const auto & item : pointValuesWithCovariance)
144  {
145  VectorType val = representer->PointSampleToPointSampleVector(item.first.second);
146  unsigned pt_id = representer->GetPointIdForPoint(item.first.first);
147 
148  // In the formulas, we actually need the precision matrix, which is the inverse of the covariance.
149  const MatrixType pointPrecisionMatrix = item.second.inverse();
150 
151  // Get the three rows pertaining to this point:
152  const MatrixType Qrows_for_pt_id = Q.block(pt_id * dim, 0, dim, numPrincipalComponents);
153 
154  Q_g.block(i * dim, 0, dim, numPrincipalComponents) = Qrows_for_pt_id;
155  mu_g.block(i * dim, 0, dim, 1) = mu.block(pt_id * dim, 0, dim, 1);
156  s_g.block(i * dim, 0, dim, 1) = val;
157 
158  LQ_g.block(i * dim, 0, dim, numPrincipalComponents) = pointPrecisionMatrix * Qrows_for_pt_id;
159  i++;
160  }
161 
162  VectorType D2 = inputModel->GetPCAVarianceVector().array();
163  const MatrixType & Q_gT = Q_g.transpose();
164 
165  MatrixType M = Q_gT * LQ_g;
166  M.diagonal() += VectorType::Ones(Q_g.cols());
167 
168  MatrixTypeDoublePrecision Minv = M.cast<double>().inverse();
169 
170  // the MAP solution for the latent variables (coefficients)
171  VectorType coeffs = Minv.cast<ScalarType>() * LQ_g.transpose() * (s_g - mu_g);
172 
173  // the MAP solution in the sample space
174  VectorType mu_c = inputModel->GetRepresenter()->SampleToSampleVector(inputModel->DrawSample(coeffs));
175 
176  const VectorType & pcaVariance = inputModel->GetPCAVarianceVector();
177  VectorTypeDoublePrecision pcaSdev = pcaVariance.cast<double>().array().sqrt();
178 
179  VectorType D2MinusRho = D2 - VectorType::Ones(D2.rows()) * rho2;
180  // the values of D2 can be negative. We need to be careful when taking the root
181  for (i = 0; i < D2MinusRho.rows(); i++)
182  {
183  D2MinusRho(i) = std::max((ScalarType)0, D2(i));
184  }
185  VectorType D2MinusRhoSqrt = D2MinusRho.array().sqrt();
186 
187 
188  using SVDType = Eigen::JacobiSVD<MatrixTypeDoublePrecision>;
189  MatrixTypeDoublePrecision innerMatrix =
190  D2MinusRhoSqrt.cast<double>().asDiagonal() * Minv * D2MinusRhoSqrt.cast<double>().asDiagonal();
191  SVDType svd(innerMatrix, Eigen::ComputeThinU);
192 
193  // SVD of the inner matrix
194  VectorType D_c = svd.singularValues().cast<ScalarType>();
195 
196  // Todo: Maybe it is possible to do this with Q, so that we don"t need to get U as well.
197  MatrixType U_c = inputModel->GetOrthonormalPCABasisMatrix() * svd.matrixU().cast<ScalarType>();
198 
199  auto PosteriorModel = StatisticalModelType::SafeCreate(representer, mu_c, U_c, D_c, rho2);
200 
201  // Write the parameters used to build the models into the builderInfo
202 
203  typename ModelInfo::BuilderInfoList builderInfoList = inputModel->GetModelInfo().GetBuilderInfoList();
204 
205  BuilderInfo::ParameterInfoList bi;
206  bi.emplace_back("NoiseVariance ", std::to_string(rho2));
207  bi.emplace_back("FixedPointsVariance ", std::to_string(0.2));
208  //
209  BuilderInfo::DataInfoList di;
210 
211  unsigned pt_no = 0;
212  for (const auto & item : pointValuesWithCovariance)
213  {
214  VectorType val = representer->PointSampleToPointSampleVector(item.first.second);
215 
216  // TODO we looked up the PointId for the same point before. Having it here again is inefficient.
217  unsigned pt_id = representer->GetPointIdForPoint(item.first.first);
218  std::ostringstream keySStream;
219  keySStream << "Point constraint " << pt_no;
220  std::ostringstream valueSStream;
221  valueSStream << "(" << pt_id << ", (";
222 
223  for (unsigned d = 0; d < dim - 1; d++)
224  {
225  valueSStream << val[d] << ",";
226  }
227  valueSStream << val[dim - 1];
228  valueSStream << "))";
229  di.emplace_back(keySStream.str(), valueSStream.str());
230  pt_no++;
231  }
232 
233  builderInfoList.emplace_back("PosteriorModelBuilder", di, bi);
234 
235  MatrixType inputScores = inputModel->GetModelInfo().GetScoresMatrix();
236  MatrixType scores = MatrixType::Zero(inputScores.rows(), inputScores.cols());
237 
238  if (computeScores)
239  {
240  // get the scores from the input model
241  for (i = 0; i < inputScores.cols(); i++)
242  {
243  // reconstruct the sample from the input model and project it back into the model
244  auto ds = inputModel->DrawSample(inputScores.col(i));
245  scores.col(i) = PosteriorModel->ComputeCoefficients(ds);
246  representer->DeleteDataset(ds);
247  }
248  }
249 
250  PosteriorModel->SetModelInfo(ModelInfo{ scores, builderInfoList });
251 
252  return PosteriorModel;
253 }
