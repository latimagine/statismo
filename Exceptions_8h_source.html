122 template <typename Callable, typename R = std::invoke_result_t<Callable>>
123 struct TranslateImpl
124 {
125  std::tuple<Status, std::optional<R>>
126  operator()(Callable && f)
127  {
128  try
129  {
130  return std::make_tuple(Status::SUCCESS, std::forward<Callable>(f)());
131  }
132  catch (const StatisticalModelException & ex)
133  {
134  return std::make_tuple(ex.GetStatus(), std::nullopt);
135  }
136  catch (const std::exception & e)
137  {
138  return std::make_tuple(Status::INTERNAL_ERROR, std::nullopt);
139  }
140  }
141 };
